13vector<ParInt> Torneo::conv(const string& res) {
14
15 vector<ParInt> v;
16 int n = res.length();
17 int x, y, cont = 0;
18
19 for (int i = 0; i < n; ++i) {
20 if (res[i] >= '0' and res[i] <= '7') {
21 if (cont == 0) x = res[i] - '0';
22 else y = res[i] - '0';
23 ++cont;
24 }
25
26 if (cont == 2) {
27 ParInt p(x,y);
28 v.push_back(p);
29 cont = 0;
30 }
31 }
32 return v;
33}

35int Torneo::match(int a, int b, const string& res) {
36
37 if (res == "1-0") return a;
38 else if (res == "0-1") return b;
39 else {
40
41 // vector de sets dels resultats
42 vector<ParInt> v = conv(res);
43 int n = v.size();
44 int x = 0, y = 0;
45
46 for (int i = 0; i < n; ++i) {
47 if (v[i].primer() > v[i].segon()) ++x;
48 else ++y;
49 }
50
51 if (x > y) return a;
52 else return b;
53 }
54}

72void Torneo::definir_emparejamientos(BinTree<int>& arb, int h, int l, int a, int x) {
73
74 // Si estamos en el ultimo nivel, tenemos un nodo sin hijos
75 if (l == h)
76 arb = BinTree<int>(a);
77
78 // Si no estamos en el ultimo nivel:
79 else {
80
81 // Si a es del nivel h-1 y a <= m-1, tenemos un nodo sin hijos
82 if (l == h-1 and a <= x)
83 arb = BinTree<int>(a);
84
85 else {

101void Torneo::proc_resultados(BinTree<int>& a, const BinTree<int>& emp, const BinTree<string>& res) {
102
103 if (emp.left().empty() and emp.right().empty()) {

107

109

123 a = BinTree<int>(x, i, d);

127void Torneo::act_stats(int a, int b, const string& res, ConjuntoJugadores& jugadores, int nivel) {
128
129 vector<ParInt> v = conv(res);
130 int n = v.size();
131 int x = 0, y = 0;
132 int juegos_a = 0, juegos_b = 0;
133
134 if (n != 1) {
135 for (int i = 0; i < n; ++i) {
136 juegos_a += v[i].primer();
137 juegos_b += v[i].segon();
138
139 if (v[i].primer() > v[i].segon()) {
140 jugadores.sumar_restar_set(a, b, jug_edicion_actual);
141 ++x;
142 } else {
143 jugadores.sumar_restar_set(b, a, jug_edicion_actual);
144 ++y;
145 }
146 }
147
148 jugadores.sumar_restar_juegos(a, b, juegos_a, juegos_b, jug_edicion_actual);
149
150 if (x > y) {
151 jugadores.sumar_restar_partido(a, b, jug_edicion_actual);
152 jugadores.sumar_puntos(cat, nivel+1, b, jug_edicion_actual);
153
154 } else {
155 jugadores.sumar_restar_partido(b, a, jug_edicion_actual);
156 jugadores.sumar_puntos(cat, nivel+1, a, jug_edicion_actual);
157 }
158
159 if (nivel == 1) {
160 if (x > y)
161 jugadores.sumar_puntos(cat, 1, a, jug_edicion_actual);
162 else
163 jugadores.sumar_puntos(cat, 1, b, jug_edicion_actual);
164 }
165
166 } else {
167
168 if (v[0].primer() > v[0].segon()) {
169 jugadores.sumar_restar_partido(a, b, jug_edicion_actual);
170 jugadores.sumar_puntos(cat, nivel+1, b, jug_edicion_actual);
171 } else {
172 jugadores.sumar_restar_partido(b, a, jug_edicion_actual);
173 jugadores.sumar_puntos(cat, nivel+1, a, jug_edicion_actual);
174 }
175 if (nivel == 1) {
176 if (v[0].primer() > v[0].segon())
177 jugadores.sumar_puntos(cat, 1, a, jug_edicion_actual);
178 else
179 jugadores.sumar_puntos(cat, 1, b, jug_edicion_actual);
180 }
181
182 }
