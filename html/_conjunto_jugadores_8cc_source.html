41void ConjuntoJugadores::baja_jugador(string& id_p) {
42
43 // Eliminamos al jugador del mapa y disminuimos el numero de jugadores
44 --njug;
45 map_jug.erase(id_p);
46
47 // Buscamos el jugador que queremos eliminar en el ranking y lo intercambiamos con
48 // los jugadores posteriores subiendo la posición de estos jugadores hasta dejar
49 // al jugador eliminado ultimo y hacemos pop_back para eliminarlo
50 int n = rank.size();
51 int i = 0;
52 bool trobat = false;
53 while (i < n-1 and not trobat) {
54 if (rank[i].consultar_nombre() == id_p) {
55 trobat = true;
56 for (int j = i; j < n-1; ++j) {
57 swap(rank, j, j+1);
58 string x = rank[j].consultar_nombre();
59 map_jug[x].modificar_posicion(j+1);
60 }
61 }
62 ++i;
63 }
64 rank.pop_back();
65}
